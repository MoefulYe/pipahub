## 非递归先序遍历
```
previsit(node,visitor) {
	s := [node]
	while Some(node) = s.pop() {
		if node is null { continue }
		visitor.visit(node)
		s.push(node.right)
		s.push(node.left)
	}
}
```
## 非递归中序遍历
```
invisit(node, visitor) {
	s := []
	cur := node
	while s.not_empty() or cur !== null {
		if cur !== null {
			s.push(cur)
			cur = cur.left
		} else {
			node = s.pop()
			visitor.visit(node)
			cur = node.right
		}
	}
}
```
## 层次遍历
```
levelorder(node, visitor) {
	q := [node]
	while Some(node) = q.deqeque() {
		if node is null { continue }
			visitor.visit(node)
			q.enqueue(node.left)
			q.enqueue(node.right)
	}
}
```
